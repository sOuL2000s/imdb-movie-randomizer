        // Function to shuffle an array (Fisher-Yates algorithm)
        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]]; // Swap elements
            }
            return array;
        }

        // Function to generate and display random movies
        function generateRandomMovies() {
            const numMoviesInput = document.getElementById('numMovies');
            let count = parseInt(numMoviesInput.value, 10);

            // Validate input: ensure count is a number, within bounds
            if (isNaN(count) || count < 1) {
                count = 1; // Default to 1 if invalid
                numMoviesInput.value = 1;
            } else if (count > allMovies.length) {
                count = allMovies.length; // Max out at available movies
                numMoviesInput.value = allMovies.length;
            }

            // Create a shallow copy of the movies array to shuffle without modifying the original
            const shuffledMovies = shuffleArray([...allMovies]); 
            const selectedMovies = shuffledMovies.slice(0, count);

            displayMovies(selectedMovies);
        }

        // Function to display movies on the page
        function displayMovies(movies) {
            const movieListElement = document.getElementById('movieList');
            movieListElement.innerHTML = ''; // Clear previous results

            if (movies.length === 0) {
                movieListElement.innerHTML = '<li class="movie-item"><div class="details">No movies to display.</div></li>';
                return;
            }

            movies.forEach((movie, index) => {
                const listItem = document.createElement('li');
                listItem.classList.add('movie-item');
                listItem.innerHTML = `
                    <div class="rank">${index + 1}.</div>
                    <div class="details">
                        <h3>${movie.title} (${movie.year})</h3>
                        <p>Rating: ${movie.rating} (${movie.votes} votes)</p>
                    </div>
                `;
                movieListElement.appendChild(listItem);
            });
        }

        // Event Listener for the button
        document.getElementById('generateBtn').addEventListener('click', generateRandomMovies);

        // Generate some random movies on initial page load
        generateRandomMovies();
